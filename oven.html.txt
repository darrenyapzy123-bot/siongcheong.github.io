<script>
    // ------------------- 菜单 JS 逻辑 (完全复制自 freezer (1).html) -------------------
    function toggleNav() {
        const sidebar = document.getElementById("mySidebar");
        const overlay = document.getElementById("overlay");
        const body = document.body;

        const isOpen = sidebar.offsetWidth > 0;

        if (isOpen) {
            sidebar.style.width = "0";
            overlay.style.display = "none";
            body.style.overflow = "auto";
        } else {
            // Ensure other submenus are closed before opening
            document.querySelectorAll(".submenu").forEach(s => (s.style.display = "none"));
            sidebar.style.width = "250px"; 
            overlay.style.display = "block";
            body.style.overflow = "hidden"; 
        }
    }
    window.toggleNav = toggleNav; 

    function toggleSubmenu(element) {
        const submenu = element.nextElementSibling;
        submenu.style.display = (submenu.style.display === 'block') ? 'none' : 'block';
    }
    window.toggleSubmenu = toggleSubmenu; 

    // ------------------- 筛选 JS 逻辑 (修改以处理容量和标题) -------------------
    document.addEventListener('DOMContentLoaded', function() {
        const brandFilter = document.getElementById('brand-filter-dropdown');
        const capacityFilter = document.getElementById('capacity-filter-dropdown');
        const allProducts = document.querySelectorAll('.product-card');
        const allTitles = document.querySelectorAll('.brand-group-title');

        // Combined filter event listener
        brandFilter.addEventListener('change', applyFilters);
        capacityFilter.addEventListener('change', applyFilters);

        window.applyFilters = function() {
            const selectedBrand = brandFilter.value;
            const selectedCapacity = capacityFilter.value;
            let visibleBrands = new Set();
            
            allProducts.forEach(card => {
                const cardBrand = card.getAttribute('data-brand');
                const cardCapacity = card.getAttribute('data-capacity'); // Get capacity data

                const brandMatch = (selectedBrand === 'all' || cardBrand === selectedBrand);
                const capacityMatch = (selectedCapacity === 'all' || cardCapacity === selectedCapacity);
                
                if (brandMatch && capacityMatch) {
                    card.style.display = 'flex';
                    visibleBrands.add(cardBrand);
                } else {
                    card.style.display = 'none';
                }
            });

            // Filter Brand Group Titles
            allTitles.forEach(title => {
                const titleBrand = title.getAttribute('data-brand');
                
                if (visibleBrands.has(titleBrand)) {
                    title.style.display = 'block';
                } else {
                    title.style.display = 'none';
                }
            });
        }
        
        // --- Back to Home (Scroll) Logic ---
        const backToTopBtn = document.getElementById("backToTopBtn");
        
        window.onscroll = function() { scrollFunction() };

        function scrollFunction() {
            const scrollTop = window.scrollY || document.documentElement.scrollTop;
            const maxScroll = document.documentElement.scrollHeight - window.innerHeight;
            const nearBottomThreshold = 50; 
            
            // Show button only when near the bottom
            if (maxScroll - scrollTop <= nearBottomThreshold && maxScroll > 0) {
                backToTopBtn.style.display = "block";
            } else {
                backToTopBtn.style.display = "none";
            }
        }

        function topFunction() {
            window.scrollTo({
                top: 0,
                behavior: 'smooth'
            });
        }
        window.topFunction = topFunction; 

        // Apply filters on initial load
        window.applyFilters();
    });
</script>
