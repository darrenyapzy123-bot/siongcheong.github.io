<script>
    document.addEventListener('DOMContentLoaded', function() {
        const brandFilter = document.getElementById('brand-filter');
        const hpFilter = document.getElementById('hp-filter');
        const typeFilter = document.getElementById('type-filter');
        const productList = document.getElementById('product-list');
        const cards = productList.querySelectorAll('.product-card');

        // 绑定筛选事件
        brandFilter.addEventListener('change', applyFilters);
        hpFilter.addEventListener('change', applyFilters);
        typeFilter.addEventListener('change', applyFilters);

        // 筛选函数：根据当前选择的品牌、HP和类型过滤卡片
        function applyFilters() {
            const selectedBrand = brandFilter.value;
            const selectedHP = hpFilter.value;
            const selectedType = typeFilter.value;

            cards.forEach(card => {
                const cardBrand = card.getAttribute('data-brand');
                const cardHP = card.getAttribute('data-hp');
                const cardType = card.getAttribute('data-type');

                const brandMatch = (selectedBrand === 'all' || selectedBrand === cardBrand);
                const hpMatch = (selectedHP === 'all' || selectedHP === cardHP);
                const typeMatch = (selectedType === 'all' || selectedType === cardType);

                if (brandMatch && hpMatch && typeMatch) {
                    card.classList.remove('hidden');
                } else {
                    card.classList.add('hidden');
                }
            });
        }

        // --- Back to Home 按钮逻辑 ---
        const backToHomeButton = document.getElementById('back-to-home');
        
        window.addEventListener('scroll', function() {
            // 当滚动到底部时显示按钮。这里用滚动条位置与文档总高度差的阈值来判断。
            const scrollThreshold = 300; // 滚动超过 300px 时显示
            const isScrolledDown = window.scrollY > scrollThreshold;

            if (isScrolledDown) {
                backToHomeButton.classList.add('show');
            } else {
                backToHomeButton.classList.remove('show');
            }
        });

        // --- 菜单和侧边栏逻辑 ---
        window.toggleNav = function() {
            const sidebar = document.getElementById("mySidebar");
            const overlay = document.getElementById("overlay");
            const body = document.body;
            
            const isOpen = sidebar.offsetWidth > 0;

            if (isOpen) {
                sidebar.style.width = "0";
                overlay.style.display = "none";
                body.style.overflow = "auto";
            } else {
                sidebar.style.width = "250px";
                overlay.style.display = "block";
                body.style.overflow = "hidden";
            }
        }

        window.toggleSubmenu = function(element) {
            const submenu = element.nextElementSibling;
            const isShown = submenu.style.display === "block";

            document.querySelectorAll(".submenu").forEach(s => (s.style.display = "none"));

            submenu.style.display = isShown ? "none" : "block";
        }
        
        // 确保页面加载时应用一次筛选（如果有默认值的话）
        applyFilters();

        // 隐藏加载失败的图片，避免破图影响布局 (防御性 JS)
        document.querySelectorAll('.product-card img').forEach(img => {
            img.onerror = function() { 
                // 仅显示一个通用的图片占位符（如果第一个也失败了，就使用备用的）
                this.src='placeholder_ac_fallback.png'; 
                this.onerror = null; // 防止无限循环
            };
        });
    });
</script>
