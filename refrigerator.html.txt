    // ------------------- 产品渲染和筛选逻辑 -------------------

    function renderProducts(productsToRender = refrigeratorProducts) {
        const productList = document.getElementById('product-list');
        const noProducts = document.getElementById('no-products');
        productList.innerHTML = ''; // 清空现有内容

        if (productsToRender.length === 0) {
            productList.style.display = 'none';
            noProducts.style.display = 'block';
            return;
        }

        productsToRender.forEach(product => {
            const card = document.createElement('div');
            card.className = 'product-card';
            
            // 默认图片如果不存在，使用一个通用占位图
            const imageSrc = product.image || 'default_refrigerator.jpg';

            card.innerHTML = `
                <img src="${imageSrc}" alt="${product.model}">
                <div class="product-info">
                    <h3>${product.model} (${product.liters}L)</h3>
                    <p>${product.description}</p>
                    <p style="font-size: 0.85em; color: #555; text-align: left;">
                        ${product.details}
                    </p>
                    <a class="btn-detail" href="#${product.model.replace(/\s/g, '_')}">查看更多</a>
                </div>
            `;
            productList.appendChild(card);
        });

        productList.style.display = 'grid';
        noProducts.style.display = 'none';
    }

    function filterProducts() {
        const selectedCategory = document.getElementById('capacity-filter').value;

        if (selectedCategory === 'all') {
            renderProducts(refrigeratorProducts);
            return;
        }

        const filtered = refrigeratorProducts.filter(product => {
            // 根据容量字段进行筛选
            if (selectedCategory === 'mini') {
                return product.liters < 200;
            } else if (selectedCategory === 'medium') {
                return product.liters >= 200 && product.liters < 400;
            } else if (selectedCategory === 'large') {
                return product.liters >= 400;
            }
            return false;
        });

        renderProducts(filtered);
    }
