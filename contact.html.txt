<script>
    // 侧边栏打开/关闭逻辑 (从 index.html 引入)
    window.toggleNav = function() {
        const sidebar = document.getElementById("mySidebar");
        const overlay = document.getElementById("overlay");
        const body = document.body;
        const mainContent = document.getElementById("main-content"); 
        
        // 假设滚动条宽度约为 17px 
        const scrollBarWidth = '17px'; 
        
        // 检查侧边栏是否已打开
        const isOpened = sidebar.style.width === '250px';

        if (isOpened) {
            // 关闭菜单
            sidebar.style.width = '0';
            overlay.style.display = 'none';
            body.style.overflow = 'auto'; // 恢复滚动
            mainContent.style.marginRight = '0'; // 移除补偿边距
        } else {
            // 打开菜单
            // 添加 margin-right 补偿，防止页面内容跳动
            mainContent.style.marginRight = scrollBarWidth;
            
            body.style.overflow = 'hidden'; // 禁止滚动
            sidebar.style.width = '250px';
            overlay.style.display = 'block';
        }
    }

    // 子菜单展开/收起逻辑 (从 index.html 引入)
    window.toggleSubmenu = function(element) {
        // 获取紧随其后的 .submenu 元素
        const submenu = element.nextElementSibling;
        
        if (submenu.style.display === "block") {
            submenu.style.display = "none";
        } else {
            // 先关闭所有其他打开的子菜单 (可选，但推荐)
            document.querySelectorAll('.submenu').forEach(sub => {
                if (sub !== submenu) {
                    sub.style.display = 'none';
                }
            });
            submenu.style.display = "block";
        }
    }
</script>
